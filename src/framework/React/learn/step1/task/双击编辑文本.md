```tsx
/**
 * 双击编辑文本 - 零闪烁版
 * 要点：
 * 1. 只用一次 useState，避免多次 setState 造成中间空白（批处理）
 * 2. 用函数式更新保证快速双击不丢值
 * 3. React 17+ 已移除事件池，事件对象可放心异步使用
 * 4. 用合成事件 onBlur/onKeyDown 替代 setTimeout，杜绝闪动
 */
import React, { useState, useRef, useEffect } from 'react';

export default function InlineEdit({ defaultValue = '双击编辑我' }) {
  // 唯一状态：编辑模式/文本
  const [edit, setEdit] = useState({ editing: false, text: defaultValue });

  // 聚焦用
  const inputRef = useRef(null);

  // 进入编辑
  const startEdit = () =>
    setEdit(prev => (prev.editing ? prev : { ...prev, editing: true }));

  // 退出并保存
  const finish = e =>
    setEdit(prev => ({ editing: false, text: e.target.value.trim() || prev.text }));

  // 按键拦截：Enter 保存，Escape 取消
  const onKey = e => {
    if (e.key === 'Enter') finish(e);
    if (e.key === 'Escape') setEdit(prev => ({ ...prev, editing: false }));
  };

  // 切到编辑模式后自动聚焦，避免用户再点一次
  useEffect(() => {
    if (edit.editing) inputRef.current?.focus();
  }, [edit.editing]);

  return edit.editing ? (
    <input
      ref={inputRef}
      defaultValue={edit.text}
      onBlur={finish}
      onKeyDown={onKey}
    />
  ) : (
    <span onDoubleClick={startEdit} >
      {edit.text}
    </span>
  );
}
```
