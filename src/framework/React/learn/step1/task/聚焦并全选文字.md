// FocusSelectInput.tsx
import { forwardRef, useRef, useImperativeHandle, useEffect } from 'react';

export interface FocusSelectInputRef {
  focus: () => void;
}

const FocusSelectInput = forwardRef<FocusSelectInputRef, {}>(
  function FocusSelectInput(props, ref) {
    const inputRef = useRef<HTMLInputElement>(null);

    // 1. 只暴露 focus 命令
    useImperativeHandle(ref, () => ({
      focus: () => {
        const node = inputRef.current;
        if (!node) return;
        node.focus();
        node.select(); // 聚焦并全选
      },
    }));

    // 2. 客户端自动执行一次（可选）
    useEffect(() => {
      inputRef.current?.focus();
      inputRef.current?.select();
    }, []);

    return <input {...props} ref={inputRef} />;
  }
);

export default FocusSelectInput;

// Demo.tsx
import { useRef } from 'react';
import FocusSelectInput, { FocusSelectInputRef } from './FocusSelectInput';

function Demo() {
  const inputRef = useRef<FocusSelectInputRef>(null);

  return (
    <>
      <button onClick={() => inputRef.current?.focus()}>聚焦+全选</button>
      <FocusSelectInput ref={inputRef} defaultValue="双击我会全选" />
    </>
  );
}
