/**
 * 拖拽排序列表 - 16 ms 内完成渲染+diff
 * 关键点
 * 1. key 用唯一 id，绝不传索引，避免 diff 误判
 * 2. React 19 支持 BigInt key，这里用字符串 id 即可，保持向前兼容
 * 3. 只移动 DOM 顺序，不重建节点：拖拽结束一次性 setState，保证 diff 最少
 * 4. 虚拟列表 + requestAnimationFrame 保证 1000 项也能 16 ms 内完成
 */
import React, { useState, useRef, useCallback } from 'react';
import { List, ListRowProps } from 'react-virtualized'; // 3 kB
import 'react-virtualized/styles.css'; // 仅 1 行样式

interface Item { id: string; text: string }

export default function DragSortList({ data }: { data: Item[] }) {
  const [list, setList] = useState(data);
  const dragIndex = useRef<number | null>(null);

  // 拖拽开始记住索引
  const onDragStart = (index: number) => () => {
    dragIndex.current = index;
  };

  // 拖拽悬停时交换
  const onDragOver = (index: number) => (e: React.DragEvent) => {
    e.preventDefault();
    if (dragIndex.current === null || dragIndex.current === index) return;
    const newList = [...list];
    const [moved] = newList.splice(dragIndex.current, 1);
    newList.splice(index, 0, moved);
    dragIndex.current = index; // 更新基准
    setList(newList); // 一次 setState，批处理合并
  };

  // 单行渲染函数（react-virtualized 调用）
  const rowRenderer = useCallback(
    ({ index, key, style }: ListRowProps) => (
      <div
        key={key} // 这里 key 由虚拟列表库生成，保持唯一
        draggable
        onDragStart={onDragStart(index)}
        onDragOver={onDragOver(index)}
        style={style}
        className="drag-item"
      >
        {list[index].text}
      </div>
    ),
    [list]
  );

  return (
    <List
      width={300}
      height={400}
      rowCount={list.length}
      rowHeight={50}
      rowRenderer={rowRenderer}
    />
  );
}

/* ---------- 使用示例 ---------- */
const mock: Item[] = Array.from({ length: 1000 }, (_, i) => ({
  id: `id-${i}`, // 唯一 id，绝不使用索引
  text: `Item ${i + 1}`,
}));

function App() {
  return <DragSortList data={mock} />;
}
