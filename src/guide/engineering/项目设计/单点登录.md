# 单点登录

单点登录（Single Sign-On，SSO）是一种身份验证机制，用户只需要登录一次就能访问多个相关但独立的系统。比如登录了 Google 账号后，可以直接使用 Gmail、YouTube、Google Drive 等服务，不需要重复输入密码。

前端实现 SSO 主要有三种方式：

### 1）同域名下的 SSO

通过共享 Cookie 实现。所有子系统使用相同的父域名，将认证信息存储在父域名的 Cookie 中。

```javascript
// 设置父域名 Cookie
document.cookie = "token=abc123; domain=.example.com; path=/";

// 子系统读取 Cookie
function getToken() {
  const cookies = document.cookie.split(';');
  for (let cookie of cookies) {
    const [name, value] = cookie.trim().split('=');
    if (name === 'token') {
      return value;
    }
  }
  return null;
}
```

### 2）跨域 SSO - CAS 模式

通过中央认证服务器（CAS）实现跨域单点登录。

```javascript
// 检查用户是否已登录
function checkLogin() {
  const token = localStorage.getItem('sso_token');
  if (!token) {
    // 重定向到 CAS 登录页面
    const returnUrl = encodeURIComponent(window.location.href);
    window.location.href = `https://sso.example.com/login?return=${returnUrl}`;
    return;
  }
  
  // 验证 token 有效性
  validateToken(token);
}

// 处理 CAS 回调
function handleCASCallback() {
  const urlParams = new URLSearchParams(window.location.search);
  const ticket = urlParams.get('ticket');
  
  if (ticket) {
    // 向 CAS 服务器验证票据
    fetch(`https://sso.example.com/validate?ticket=${ticket}`)
      .then(response => response.json())
      .then(data => {
        if (data.valid) {
          localStorage.setItem('sso_token', data.token);
          // 清除 URL 中的票据参数
          window.history.replaceState({}, '', window.location.pathname);
        }
      });
  }
}
```

### 3）基于 JWT 的 SSO

使用 JSON Web Token 在多个系统间传递用户信息。

```javascript
// JWT 工具函数
function parseJWT(token) {
  try {
    const base64Url = token.split('.')[1];
    const base64 = base64Url.replace(/-/g, '+').replace(/_/g, '/');
    const jsonPayload = decodeURIComponent(
      atob(base64).split('').map(c => 
        '%' + ('00' + c.charCodeAt(0).toString(16)).slice(-2)
      ).join('')
    );
    return JSON.parse(jsonPayload);
  } catch (error) {
    return null;
  }
}

// 检查 JWT 是否过期
function isTokenExpired(token) {
  const payload = parseJWT(token);
  if (!payload || !payload.exp) return true;
  
  const currentTime = Math.floor(Date.now() / 1000);
  return payload.exp < currentTime;
}

// SSO 登录流程
class SSOManager {
  constructor(ssoServer) {
    this.ssoServer = ssoServer;
    this.token = localStorage.getItem('sso_token');
  }
  
  async checkAuth() {
    if (!this.token || isTokenExpired(this.token)) {
      await this.redirectToLogin();
      return false;
    }
    return true;
  }
  
  async redirectToLogin() {
    const returnUrl = encodeURIComponent(window.location.href);
    window.location.href = `${this.ssoServer}/login?return=${returnUrl}`;
  }
  
  handleCallback() {
    const urlParams = new URLSearchParams(window.location.search);
    const token = urlParams.get('token');
    
    if (token) {
      localStorage.setItem('sso_token', token);
      window.history.replaceState({}, '', window.location.pathname);
      return true;
    }
    return false;
  }
  
  logout() {
    localStorage.removeItem('sso_token');
    window.location.href = `${this.ssoServer}/logout`;
  }
}
```

### 2）前端 SSO 实现的关键技术点

#### PostMessage 跨域通信

当需要在不同域名的页面间传递认证信息时，可以使用 PostMessage API。

```javascript
// 父页面（SSO 认证页面）
function sendAuthResult(targetOrigin, authData) {
  window.parent.postMessage({
    type: 'SSO_AUTH_SUCCESS',
    data: authData
  }, targetOrigin);
}

// 子页面（业务应用）
window.addEventListener('message', function(event) {
  if (event.origin !== 'https://sso.example.com') return;
  
  if (event.data.type === 'SSO_AUTH_SUCCESS') {
    const { token, userInfo } = event.data.data;
    localStorage.setItem('auth_token', token);
    localStorage.setItem('user_info', JSON.stringify(userInfo));
    
    // 关闭 SSO 弹窗或 iframe
    closeAuthWindow();
  }
});
```

#### iframe 嵌入式认证

```javascript
// 创建隐藏的 iframe 进行静默认证
function silentAuth() {
  return new Promise((resolve, reject) => {
    const iframe = document.createElement('iframe');
    iframe.style.display = 'none';
    iframe.src = 'https://sso.example.com/silent-auth';
    
    const messageHandler = (event) => {
      if (event.origin !== 'https://sso.example.com') return;
      
      window.removeEventListener('message', messageHandler);
      document.body.removeChild(iframe);
      
      if (event.data.success) {
        resolve(event.data.token);
      } else {
        reject(new Error('Silent auth failed'));
      }
    };
    
    window.addEventListener('message', messageHandler);
    document.body.appendChild(iframe);
    
    // 设置超时
    setTimeout(() => {
      window.removeEventListener('message', messageHandler);
      document.body.removeChild(iframe);
      reject(new Error('Silent auth timeout'));
    }, 5000);
  });
}
```

### 3）SSO 安全考虑

#### CSRF 防护

```javascript
// 生成 CSRF Token
function generateCSRFToken() {
  return Math.random().toString(36).substring(2) + Date.now().toString(36);
}

// 在请求中携带 CSRF Token
function makeAuthenticatedRequest(url, data) {
  const csrfToken = generateCSRFToken();
  sessionStorage.setItem('csrf_token', csrfToken);
  
  return fetch(url, {
    method: 'POST',
    headers: {
      'Content-Type': 'application/json',
      'X-CSRF-Token': csrfToken,
      'Authorization': `Bearer ${localStorage.getItem('sso_token')}`
    },
    body: JSON.stringify(data)
  });
}
```

#### Token 刷新机制

```javascript
class TokenManager {
  constructor() {
    this.refreshPromise = null;
  }
  
  async getValidToken() {
    const token = localStorage.getItem('access_token');
    const refreshToken = localStorage.getItem('refresh_token');
    
    if (!token) return null;
    
    // 检查 token 是否即将过期（提前 5 分钟刷新）
    const payload = parseJWT(token);
    const expirationTime = payload.exp * 1000;
    const currentTime = Date.now();
    const fiveMinutes = 5 * 60 * 1000;
    
    if (expirationTime - currentTime < fiveMinutes) {
      return await this.refreshAccessToken(refreshToken);
    }
    
    return token;
  }
  
  async refreshAccessToken(refreshToken) {
    // 防止并发刷新
    if (this.refreshPromise) {
      return await this.refreshPromise;
    }
    
    this.refreshPromise = fetch('/api/auth/refresh', {
      method: 'POST',
      headers: { 'Content-Type': 'application/json' },
      body: JSON.stringify({ refreshToken })
    })
    .then(response => response.json())
    .then(data => {
      localStorage.setItem('access_token', data.accessToken);
      localStorage.setItem('refresh_token', data.refreshToken);
      return data.accessToken;
    })
    .finally(() => {
      this.refreshPromise = null;
    });
    
    return await this.refreshPromise;
  }
}
```

### 4）完整的 SSO 客户端实现

```javascript
class SSOClient {
  constructor(config) {
    this.ssoServer = config.ssoServer;
    this.clientId = config.clientId;
    this.redirectUri = config.redirectUri;
    this.tokenManager = new TokenManager();
  }
  
  // 初始化 SSO 客户端
  async init() {
    // 处理回调
    if (this.handleCallback()) {
      return;
    }
    
    // 尝试静默认证
    try {
      await this.silentAuth();
    } catch (error) {
      console.log('Silent auth failed:', error.message);
    }
  }
  
  // 检查用户是否已登录
  async isAuthenticated() {
    const token = await this.tokenManager.getValidToken();
    return !!token;
  }
  
  // 获取用户信息
  async getUserInfo() {
    const token = await this.tokenManager.getValidToken();
    if (!token) return null;
    
    const payload = parseJWT(token);
    return payload.user;
  }
  
  // 登录
  login() {
    const state = this.generateState();
    sessionStorage.setItem('sso_state', state);
    
    const params = new URLSearchParams({
      client_id: this.clientId,
      redirect_uri: this.redirectUri,
      response_type: 'code',
      state: state
    });
    
    window.location.href = `${this.ssoServer}/authorize?${params}`;
  }
  
  // 登出
  logout() {
    localStorage.removeItem('access_token');
    localStorage.removeItem('refresh_token');
    
    const params = new URLSearchParams({
      client_id: this.clientId,
      redirect_uri: this.redirectUri
    });
    
    window.location.href = `${this.ssoServer}/logout?${params}`;
  }
  
  // 生成随机状态值
  generateState() {
    return Math.random().toString(36).substring(2) + Date.now().toString(36);
  }
  
  // 处理 SSO 回调
  handleCallback() {
    const urlParams = new URLSearchParams(window.location.search);
    const code = urlParams.get('code');
    const state = urlParams.get('state');
    
    if (!code) return false;
    
    // 验证 state 参数
    const savedState = sessionStorage.getItem('sso_state');
    if (state !== savedState) {
      throw new Error('Invalid state parameter');
    }
    
    // 交换授权码获取 token
    this.exchangeCodeForToken(code);
    return true;
  }
  
  // 交换授权码
  async exchangeCodeForToken(code) {
    const response = await fetch(`${this.ssoServer}/token`, {
      method: 'POST',
      headers: { 'Content-Type': 'application/json' },
      body: JSON.stringify({
        client_id: this.clientId,
        code: code,
        redirect_uri: this.redirectUri
      })
    });
    
    const data = await response.json();
    localStorage.setItem('access_token', data.access_token);
    localStorage.setItem('refresh_token', data.refresh_token);
    
    // 清理 URL 参数
    window.history.replaceState({}, '', window.location.pathname);
  }
}

// 使用示例
const ssoClient = new SSOClient({
  ssoServer: 'https://sso.example.com',
  clientId: 'your-client-id',
  redirectUri: window.location.origin + '/callback'
});

// 页面加载时初始化
ssoClient.init();
```
