我的解决公式

function resolveConflict(issue, person) {
    // 先处理情绪，再处理问题
    acknowledgeFeeling(person);
    
    // 聚焦问题，不针对人
    focusOnProblem(issue);
    
    // 寻找共同目标
    findCommonGoal();
    
    // 妥协或找第三方案
    return compromise || findThirdWay();
}
