    todos.value.push({
      id: Date.now(),
      text: newTodo.value.trim(),

    newTodo.value = ''
    saveTodos()
  }
}

const removeTodo = (id) => {
  todos.value = todos.value.filter(todo => todo.id !== id)

  if (saved) {
    todos.value = JSON.parse(saved)

}

onMounted(() => {
